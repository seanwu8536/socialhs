
취미 게시판 전체 조회
SELECT BB.* FROM BBS_CODE BC, B_BOARD BB 
WHERE BC.BBS_IDX = BB.BBS_IDX
AND BC.BBS_TYPE = 300
;

스터디 게시판 전체 조회
SELECT BB.* FROM BBS_CODE BC, B_BOARD BB 
WHERE BC.BBS_IDX = BB.BBS_IDX
AND BC.BBS_TYPE = 700
;

기타 게시판 전체 조회
SELECT BB.* FROM BBS_CODE BC, B_BOARD BB 
WHERE BC.BBS_IDX = BB.BBS_IDX
AND BC.BBS_TYPE = 0
;
